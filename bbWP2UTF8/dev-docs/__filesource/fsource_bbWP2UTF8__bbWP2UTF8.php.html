  268.   /**
  269.    * Converts a given column's type of a given table to its binary counterpart according to
  270.    * the list below:
  271.    *
  272.    *  CHAR -> BINARY
  273.    *  VARCHAR -> VARBINARY
  274.    *  TINYTEXT -> TINYBLOB
  275.    *  TEXT -> BLOB
  276.    *  MEDIUMTEXT -> MEDIUMBLOB
  277.    *  LONGTEXT -> LONGBLOB
  278.    *
  279.    * Columns with the type ENUM and SET will be set to character set binary
  280.    * @return string performed query
  281.    * @access private
  282.    ***/
  283.   function _binarizeTableColumn($table_name, $column) 
  284.   {
  285.     if( ! is_string($table_name) || ! is_object($column) ) { return -1; } // make sure we get the right types to work with
  286.       $wpdb = $this->_wpdb;
  287.       // text columns do not need any other processing...so we'll switch them to their binary counterparts directly
  288.       switch ($column->Type) {
  289.         case 'tinytext':
  290.           $query = sprintf("ALTER TABLE %s CHANGE %s %s TINYBLOB NOT NULL", $table_name, $column->Field, $column->Field);
  291.         break;
  292.         
  293.         case 'text':
  294.           $query = sprintf("ALTER TABLE %s CHANGE %s %s BLOB NOT NULL", $table_name, $column->Field, $column->Field);
  295.         break;
  296.       
  297.         case 'mediumtext':
  298.           $query = sprintf("ALTER TABLE %s CHANGE %s %s MEDIUMBLOB NOT NULL", $table_name, $column->Field, $column->Field);
  299.         break;
  300.         
  301.         case 'longtext':
  302.           $query = sprintf("ALTER TABLE %s CHANGE %s %s LONGBLOB NOT NULL", $table_name, $column->Field, $column->Field);
  303.         break;
  304.         
  305.         // some other column types we're interested in need some additional processing 
  306.         default:
  307.           $col_type_info = $this->_processColumnInfo($column);
  308.           if( is_array($col_type_info) ) {
  309.             switch($col_type_info[0]) {
  310.             
  311.               case 'char':
  312.                 $query = sprintf("ALTER TABLE %s CHANGE %s %s BINARY ( %d ) NOT NULL", $table_name, $column->Field, $column->Field, $col_type_info[1]);
  313.               break;
  314.                 
  315.               case 'varchar':
  316.                 $query = sprintf("ALTER TABLE %s CHANGE %s %s VARBINARY ( %d ) NOT NULL", $table_name, $column->Field, $column->Field, $col_type_info[1]);
  317.               break;
  318.               
  319.               case 'enum':
  320.                 if( $this->isColumnCharSetBinary($table_name, $column->Field) ) {
  321.                   $query = null;
  322.                 } else {
  323.                   $query = sprintf("ALTER TABLE %s CHANGE %s %s %s( %s ) CHARACTER SET binary NOT NULL", $table_name, $column->Field, $column->Field, $col_type_info[0], $col_type_info[1]);
  324.                 }
  325.               break;
  326.                 
  327.               case 'set':
  328.                 if( $this->isColumnCharSetBinary($table_name, $column->Field) ) {
  329.                   $query = null;
  330.                 } else {
  331.                   $query = sprintf("ALTER TABLE %s CHANGE %s %s %s( %s ) CHARACTER SET binary NOT NULL", $table_name, $column->Field, $column->Field, $col_type_info[0], $col_type_info[1]);
  332.                 }
  333.               break;
  334.             }
  335.           }
  336.         break;
  337.       }
  338.       
  339.       if( ! is_null($query) ) {
  340.         $result = $wpdb->query($query);
  341.         return "<span style='font-size: 50%';>Performed query: $query</span><br />\n";
  342.       }
